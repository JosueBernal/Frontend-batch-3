Ejercicio de dormir

      Inicio(dormir, despertar)
        leer dormir
        leer despertar

        Si dormir > 12
         horasDormidas = (24 - dormir) + despertar
        Sino
         horasDormidas = despertar - dormir
        fin-Si

        Escribe horasDormidas
      Fin

Ejercicio Acceso al refrigerador

        Inicio(nombre, contraseña)
            leer nombre
            leer contraseña

            nombreCorrecto = "jos"
            contraseñaCorrecta = 123

            Si nombre = nombreCorrecto and contraseña = contraseñaCorrecta
                Escribe "El nombre y la contraseña es correcta"
            Sino
                Si nombre != nombreCorrecto
                    Escribe "El nombre es incorrecto"
                Fin-Si
                Si contraseña != contraseñaCorrecta
                    Escribe "La contraseña es incorrecta"
                Fin-Si
            Fin-Si
        Fin

Ejercicio Peso

        Inicio(peso, peso_ideal, cambio_de_peso)
            leer peso
            leer peso_ideal
            leer cambio_de_peso

            semanas = 0

            Para contador = peso Hasta contador = peso_ideal Disminuye contador = contador - cambio_de_peso
                semanas = semanas + 1
            Fin-Para

            Escribe semanas
        Fin
