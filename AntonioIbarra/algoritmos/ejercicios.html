Ejercicio de dormir

          Inicio (dormir, despertar)
            leer dormir
            leer despertar
            Si dormir > 12
              HorasDormidas= (24 - dormir ) + despertar
            Sino
            HorasDormidas = despertar - dormir
            Escribe HorasDormidas
          Fin

Ejercicio de Precio de Super

          Inicio (sal,azucar,cereal,arroz,helado)
            leer sal
            leer azucar
            leer cereal
            leer arroz
            leer helado

            PrecioDeSuper = sal + azucar + cereal + arroz + helado
            Escribe PrecioDeSuper
          Fin

Ejercicio de Presupuesto

          Inicio (ingreso1,ingreso2,gasto1,gasto2,gasto3)
            leer ingreso1
            leer ingreso2
            leer gasto1
            leer gasto2
            leer gasto3

            DineroFinalMes = ingreso1 + ingreso2 - gasto1 - gasto2 - gasto3
            Escribe DineroFinalMes
          Fin

Ejercicio de Herencia

          Inicio (jose,juan,herencia)
            leer jose
            leer juan
            leer herencia
            DineroTotalJose = jose + (herencia DIV 2)
            DineroTotalJuan = juan + (herencia DIV 2)
            DineroEntreLosDos = DineroTotalJose + DineroTotalJuan
            Escribe DineroTotalJose
            Escribe DineroTotalJuan
            Escribe DineroEntreLosDos
          Fin

Ejercicio Promedio

          Inicio (examen1,examen2,examen3)
            leer examen1
            leer examen2
            leer examen3
            Promedio = (examen1 + examen2 + examen3) / 3
            SI Promedio >= 8.5 Entonces
              Escribe "Tu papa te dara un premio"
            SINO examen1 < 6 OR examen2 < 6 OR examen3 < 6
              Escribe "Tu papa te castigara"
            SINO
              Escribe "Sigue esforzandote para obtener un premio"
            Fin-SI
          Fin
